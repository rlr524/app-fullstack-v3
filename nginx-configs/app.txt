server {

	root /var/www/html;
	index index.html;

	server_name emiyanet.com;
	access_log /var/log/nginx/access.log upstreamlog;

	location / {
		proxy_set_header Upgrade $http_upgrade;
		proxy_set_header Connection "upgrade";

		proxy_pass http://nodebackend;
	}

    listen [::]:443 http2 ssl ipv6only=on; # managed by Certbot
    listen 443 http2 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/emiyanet.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/emiyanet.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}
server {
    if ($host = emiyanet.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


	listen 80 default_server;
	listen [::]:80 default_server;

	server_name emiyanet.com;
    return 404; # managed by Certbot


}
